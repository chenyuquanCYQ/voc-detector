// ==========================================
// 主流程（不變動）
// ==========================================
function runMarketResearch() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const existingData = sheet.getDataRange().getValues();
  const existingProductNames = existingData.map(row => String(row[0]).toLowerCase().trim());
  const today = new Date();
  const formattedDate = Utilities.formatDate(today, Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm");

  let allSearchContent = "";
  console.log("🚀 開始搜尋，共 " + SEARCH_TASKS.length + " 個任務...");

  SEARCH_TASKS.forEach((task, i) => {
    console.log(`  [${i+1}/${SEARCH_TASKS.length}] ${task.lang}`);
    const searchData = fetchSerperData(task.query, task.gl, task.hl);
    if (searchData) {
      allSearchContent += `\n\n=== 來自 ${task.lang} 的網頁資訊 ===\n${searchData}`;
    }
    Utilities.sleep(1000);
  });

  if (allSearchContent === "") {
    console.log("⚠️ 所有任務都沒有搜到資料。");
    return;
  }

  console.log("📦 資料收集完畢，交給 Gemini 分析...");

  const extractedProducts = analyzeWithGemini(allSearchContent);
  let addedCount = 0;

  if (extractedProducts && extractedProducts.length > 0) {
    extractedProducts.forEach(product => {
      const pName = String(product.product_name).toLowerCase().trim();
      if (!existingProductNames.includes(pName) && product.product_name !== "無" && product.product_name !== "") {
        sheet.appendRow([product.product_name, product.brand, product.features, product.url, formattedDate]);
        existingProductNames.push(pName);
        addedCount++;
      }
    });
    console.log(`✅ 完成！本次新增 ${addedCount} 筆。`);
  } else {
    console.log(`⚠️ Gemini 分析完畢，無新產品。`);
  }

  // 每次執行後，額外產出一份 RAG 關鍵字建議（寫入第二個工作表）
  generateRagKeywords(allSearchContent);
}

// ==========================================
// Serper 搜尋（不變動）
// ==========================================
function fetchSerperData(query, gl, hl) {
  const url = 'https://google.serper.dev/search';
  const payload = { "q": query, "gl": gl, "hl": hl, "num": 10 };
  const options = {
    'method': 'post',
    'headers': { 'X-API-KEY': SERPER_API_KEY, 'Content-Type': 'application/json' },
    'payload': JSON.stringify(payload),
    'muteHttpExceptions': true
  };
  try {
    const response = UrlFetchApp.fetch(url, options);
    const json = JSON.parse(response.getContentText());
    if (json.organic && json.organic.length > 0) {
      return json.organic.map(item => `標題: ${item.title}\n摘要: ${item.snippet}\n網址: ${item.link}`).join('\n\n');
    }
  } catch (e) {
    console.log(`Serper 錯誤 (${gl}): ` + e);
  }
  return null;
}

function analyzeWithGemini(searchContent) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GEMINI_API_KEY}`;

  const prompt = `你是專業的醫療器材與氣味科技產品市場分析師。
請閱讀以下來自全球多個國家（包含中國大陸）的 Google 搜尋結果，找出所有真實存在的相關硬體產品。

【納入範圍，以下全部要收錄】
1. 消費電子健康裝置：口臭偵測器、體臭感測器、個人健康監測器
2. 醫療呼氣分析設備：
   - FeNO 一氧化氮檢測（氣喘診斷，如 NIOX Vero）
   - C13 尿素呼氣試驗（幽門桿菌，如 BreathID）
   - 氫氣/甲烷/H2S 呼氣檢測（腸道菌/SIBO，如 FoodMarble）
   - 呼氣代謝體/VOC 疾病診斷（如 Owlstone Medical ReCIVA）
   - 呼氣酒精/生酮/丙酮檢測
3. 工業/環境用氣體分析設備與電子鼻
4. 中國品牌產品（包含簡體中文描述的產品，品牌名保留原文）
5. 感測器模組（可嵌入其他裝置的氣體感測元件）

【排除】純學術論文、無具體產品的評論文章、與氣味/氣體完全無關的內容。

搜尋結果總匯：
${searchContent}

請嚴格使用以下 JSON 陣列格式，不含任何其他文字或 Markdown：
[
  {
    "product_name": "產品名稱（保留原文並附繁體中文翻譯，中國品牌保留簡體原名）",
    "brand": "品牌或公司名稱（保留原文）",
    "features": "一句話說明核心功能與檢測標的（繁體中文）",
    "url": "來源網址"
  }
]
找不到任何產品請回覆 []`;

  const payload = {
    "contents": [{"parts": [{"text": prompt}]}],
    "generationConfig": { "response_mime_type": "application/json" }
  };
  const options = {
    'method': 'post',
    'headers': {'Content-Type': 'application/json'},
    'payload': JSON.stringify(payload),
    'muteHttpExceptions': true
  };

  try {
    const response = UrlFetchApp.fetch(url, options);
    const resultJson = JSON.parse(response.getContentText());
    if (resultJson.error) {
      console.log(`❌ Gemini 錯誤: [${resultJson.error.code}] ${resultJson.error.message}`);
      return [];
    }
    if (!resultJson.candidates) return [];
    return JSON.parse(resultJson.candidates[0].content.parts[0].text);
  } catch (e) {
    console.log(`⚠️ 執行錯誤: ${e}`);
    return [];
  }
}

// RAG 關鍵字知識圖產生器（新增）
// 每次執行後自動更新「RAG關鍵字」工作表
// ==========================================
function generateRagKeywords(searchContent) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let ragSheet = ss.getSheetByName("RAG關鍵字");
  if (!ragSheet) {
    ragSheet = ss.insertSheet("RAG關鍵字");
  }

  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GEMINI_API_KEY}`;

  const prompt = `根據以下氣味/氣體檢測產業的搜尋資料，產出一份供 RAG（知識圖譜）使用的關鍵字分類清單。
請以 JSON 格式輸出，涵蓋以下六個維度，每個維度列出 10–20 個關鍵詞（中英文混合，保留專業術語原文）：

搜尋資料：
${searchContent.substring(0, 8000)}

請輸出以下 JSON 結構：
{
  "技術原理": ["MOS", "MEMS", "GC-MS", "NDIR", "FeNO", "C13-UBT", ...],
  "目標氣體分子": ["H2S", "NH3", "VOCs", "NO", "H2", "CH4", "丙酮", "乙醇", ...],
  "應用場景": ["口臭檢測", "氣喘診斷", "幽門桿菌", "SIBO腸道", "生酮監測", "工業安全", ...],
  "品牌與公司": ["TANITA", "Owlstone Medical", "FoodMarble", "Shownovo", ...],
  "搜尋關鍵字建議": ["breath analyzer", "ブレスチェッカー", "口气检测仪", "呼気分析", ...],
  "新興趨勢": ["呼氣代謝體", "breath biopsy", "wearable odor", "AI氣味辨識", ...]
}`;

  const payload = {
    "contents": [{"parts": [{"text": prompt}]}],
    "generationConfig": { "response_mime_type": "application/json" }
  };
  const options = {
    'method': 'post',
    'headers': {'Content-Type': 'application/json'},
    'payload': JSON.stringify(payload),
    'muteHttpExceptions': true
  };

  try {
    const response = UrlFetchApp.fetch(url, options);
    const resultJson = JSON.parse(response.getContentText());
    if (!resultJson.candidates) return;

    const keywords = JSON.parse(resultJson.candidates[0].content.parts[0].text);
    const today = new Date();
    const formattedDate = Utilities.formatDate(today, Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm");

    ragSheet.clearContents();
    ragSheet.appendRow(["維度", "關鍵詞", "更新時間"]);

    Object.entries(keywords).forEach(([dimension, words]) => {
      words.forEach(word => {
        ragSheet.appendRow([dimension, word, formattedDate]);
      });
    });

    console.log(`📚 RAG 關鍵字已更新至「RAG關鍵字」工作表，共 ${Object.values(keywords).flat().length} 筆。`);
  } catch (e) {
    console.log(`⚠️ RAG 關鍵字產生失敗: ${e}`);
  }
}
